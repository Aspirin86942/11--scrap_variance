# Output Query Diagnostics Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Align Node benchmark and WPS performance diagnostics with the current three output-sheet query paths without changing business output.

**Architecture:** Add a pure `src/core/output-query-runner.ts` layer that selects the real output-sheet path and records stable metrics. Keep WPS side effects in macro files, and make benchmark and diagnostics call the same runner used by formal current-sheet queries.

**Tech Stack:** TypeScript strict mode, Vitest, esbuild, WPS JS runtime, npm.

---

## File Structure

- Create: `src/core/output-query-runner.ts`
  - Pure business runner for `variance_summary`, `oa_doc_compare`, and `erp_doc_compare`.
  - No WPS `Range`, `Sheet`, metadata, clearing, or writing.
  - Records metrics with output-scoped stage names and notes.

- Create: `tests/core/output-query-runner.test.ts`
  - Unit tests for output matrix results, no-result messages, row counts, metric names, and metric notes.

- Modify: `src/macros/current-sheet-query.ts`
  - Remove duplicated output-sheet result builders.
  - Call `runOutputSheetQueryCore()` after source rows are parsed.
  - Preserve existing WPS behavior: only active sheet writes, metadata, query state, and error rows.

- Modify: `src/bench/query-benchmark.ts`
  - Change benchmark report shape from one core pipeline result per dataset to three output results per dataset.
  - Render output kind in the table.

- Modify: `tests/bench/query-benchmark.test.ts`
  - Update report and table expectations for output grouping.

- Modify: `src/macros/performance-diagnostics.ts`
  - Read and parse source sheets once.
  - Run all three output kinds through the pure runner.
  - Write result-scale rows per output kind.

- Modify: `tests/macros/macro-flow.test.ts`
  - Update WPS diagnostics expectations to the new output-sheet stages and result rows.

- Modify: `main.js`
  - Regenerated by `npm run build`; do not edit by hand.

```ts
import { buildDepartmentVarianceSummaryRows, departmentVarianceSummaryRowsToValues } from "./department-variance-summary";
import { buildErpDocCompare, buildOaDocCompare, docCompareRowsToValues } from "./doc-compare";
import { parseQueryDirection, QUERY_DIRECTIONS } from "./query-direction";
import { parseFilters } from "./build-oa-rows";
import type { MetricsRecorder } from "../perf/metrics";
import type { DocCompareResult, OutputMatrix, OutputSheetKind, RawRow, RibbonQueryState } from "../types/scrap";

export type RunnableOutputSheetKind = OutputSheetKind;

export interface OutputQueryRunnerInput {
  kind: RunnableOutputSheetKind;
  oaRows: RawRow[];
  erpRows: RawRow[];
  queryState: RibbonQueryState;
  metrics: MetricsRecorder;
}

export interface OutputQueryRowCounts {
  sourceRows: number;
  outputRows: number;
  summaryRows?: number;
  materialRows?: number;
}

export interface OutputQueryRunnerResult {
  kind: RunnableOutputSheetKind;
  values: OutputMatrix | null;
  noResultMessage: string | null;
  rowCounts: OutputQueryRowCounts;
}

function countMaterialRows(result: DocCompareResult): number {
  let count = 0;
  for (const rows of result.materialRowsBySummaryKey.values()) {
    count += rows.length;
  }
  return count;
}

function outputNote(kind: RunnableOutputSheetKind): string {
  return `output=${kind}`;
}

function noResult(
  kind: RunnableOutputSheetKind,
  message: string,
  sourceRows: number,
  rowCounts: Pick<OutputQueryRowCounts, "summaryRows" | "materialRows"> = {}
): OutputQueryRunnerResult {
  return {
    kind,
    values: null,
    noResultMessage: message,
    rowCounts: {
      sourceRows,
      outputRows: 1,
      ...rowCounts
    }
  };
}

function success(
  kind: RunnableOutputSheetKind,
  values: OutputMatrix,
  sourceRows: number,
  rowCounts: Pick<OutputQueryRowCounts, "summaryRows" | "materialRows">
): OutputQueryRunnerResult {
  return {
    kind,
    values,
    noResultMessage: null,
    rowCounts: {
      sourceRows,
      outputRows: values.length,
      ...rowCounts
    }
  };
}

export function runOutputSheetQueryCore(input: OutputQueryRunnerInput): OutputQueryRunnerResult {
  const filters = parseFilters(input.queryState);
  const queryDirection = parseQueryDirection(input.queryState.queryDirection);
  const sourceRows = input.oaRows.length + input.erpRows.length;

  if (input.kind === "variance_summary") {
    const summaryRows = input.metrics.measure(
      "build_variance_summary_rows",
      {
        inputRows: sourceRows,
        outputRows: (rows) => rows.length,
        note: outputNote(input.kind)
      },
      () => buildDepartmentVarianceSummaryRows(input.oaRows, input.erpRows, filters, queryDirection)
    );

    if (summaryRows.length === 0) {
      return noResult(
        input.kind,
        queryDirection === QUERY_DIRECTIONS.erpSourceToOa
          ? "查询条件没有匹配到 ERP 数据。"
          : "查询条件没有匹配到 OA 数据。",
        sourceRows,
        { summaryRows: 0 }
      );
    }

    const values = input.metrics.measure(
      "build_variance_summary_matrix",
      {
        inputRows: summaryRows.length,
        outputRows: (matrix) => matrix.length,
        note: outputNote(input.kind)
      },
      () => departmentVarianceSummaryRowsToValues(summaryRows)
    );

    return success(input.kind, values, sourceRows, { summaryRows: summaryRows.length });
  }

  if (input.kind === "oa_doc_compare") {
    const result = input.metrics.measure(
      "build_oa_doc_compare_rows",
      {
        inputRows: sourceRows,
        outputRows: (value) => value.summaryRows.length,
        note: outputNote(input.kind)
      },
      () => buildOaDocCompare(input.oaRows, input.erpRows, filters)
    );
    const materialRows = countMaterialRows(result);

    if (result.summaryRows.length === 0) {
      return noResult(input.kind, "查询条件没有匹配到 OA 数据。", sourceRows, {
        summaryRows: 0,
        materialRows
      });
    }

    const values = input.metrics.measure(
      "build_oa_doc_compare_matrix",
      {
        inputRows: result.summaryRows.length,
        outputRows: (matrix) => matrix.length,
        note: outputNote(input.kind)
      },
      () => docCompareRowsToValues("oa_doc_compare", result.summaryRows)
    );

    return success(input.kind, values, sourceRows, {
      summaryRows: result.summaryRows.length,
      materialRows
    });
  }

  const result = input.metrics.measure(
    "build_erp_doc_compare_rows",
    {
      inputRows: sourceRows,
      outputRows: (value) => value.summaryRows.length,
      note: outputNote(input.kind)
    },
    () => buildErpDocCompare(input.oaRows, input.erpRows, filters)
  );
  const materialRows = countMaterialRows(result);

  if (result.summaryRows.length === 0) {
    return noResult(input.kind, "查询条件没有匹配到 ERP 数据。", sourceRows, {
      summaryRows: 0,
      materialRows
    });
  }

  const values = input.metrics.measure(
    "build_erp_doc_compare_matrix",
    {
      inputRows: result.summaryRows.length,
      outputRows: (matrix) => matrix.length,
      note: outputNote(input.kind)
    },
    () => docCompareRowsToValues("erp_doc_compare", result.summaryRows)
  );

  return success(input.kind, values, sourceRows, {
    summaryRows: result.summaryRows.length,
    materialRows
  });
}
```

export function renderBenchTable(report: BenchReport): string {
  const lines = ["dataset     output           stage                         input_rows time_ms   heap_delta_mb_or_max"];
  for (const dataset of report.datasets) {
    for (const output of dataset.outputs) {
      for (const stage of output.stages) {
        lines.push(
          `${dataset.name.padEnd(11)} ${output.kind.padEnd(16)} ${stage.name.padEnd(29)} ${String(stage.inputRows).padEnd(9)} ${String(stage.timeMs).padEnd(9)} ${String(stage.heapDeltaMb)}`
        );
