# Query Dialog Autocomplete Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add autocomplete dropdown suggestions for `公司简称`, `一级部门`, and `二级部门` in the existing WPS query dialog while keeping free text input and current query semantics.

**Architecture:** Parent TypeScript code reads OA/ERP source sheets once before opening the dialog, builds a small token-scoped suggestions payload, and writes it to `PluginStorage` with the existing initial state. The static WPS dialog reads that payload, renders a lightweight custom dropdown with plain DOM APIs, and submits the same text fields as before.

**Tech Stack:** TypeScript, static browser-style JavaScript for WPS dialog, Vitest, Node `vm` for static dialog helper tests, npm/esbuild committed `main.js`.

```ts
import {
  ERP_REQUIRED_HEADERS,
  MAX_HEADER_SCAN_ROWS,
  MIN_ERP_HEADER_MATCH_COUNT,
  MIN_OA_HEADER_MATCH_COUNT,
  OA_REQUIRED_HEADERS,
  SHEET_NAMES
} from "../constants";
import type { RawRow } from "../types/scrap";
import type { ScrapVarianceGlobal } from "../types/wps";
import { readSheetTable } from "../wps-api/read-sheet-data";
import { getSheetByName } from "../wps-api/workbook";
import { normalizeText } from "../utils/text";

export interface QueryDialogSuggestions {
  company: string[];
  dept1: string[];
  dept2: string[];
}

export const EMPTY_QUERY_DIALOG_SUGGESTIONS: QueryDialogSuggestions = {
  company: [],
  dept1: [],
  dept2: []
};

function pickColumnText(rows: RawRow[], header: string): string[] {
  return rows.map((row) => normalizeText(row[header])).filter((value) => value.length > 0);
}

function uniqueSorted(values: string[]): string[] {
  return [...new Set(values)].sort((left, right) => left.localeCompare(right, "zh-CN"));
}

export function buildQueryDialogSuggestions(root?: ScrapVarianceGlobal): QueryDialogSuggestions {
  try {
    const oaSheet = getSheetByName(SHEET_NAMES.oa, root);
    const erpSheet = getSheetByName(SHEET_NAMES.erp, root);
    const oaTable = readSheetTable(oaSheet, [...OA_REQUIRED_HEADERS], MIN_OA_HEADER_MATCH_COUNT, MAX_HEADER_SCAN_ROWS);
    const erpTable = readSheetTable(
      erpSheet,
      [...ERP_REQUIRED_HEADERS],
      MIN_ERP_HEADER_MATCH_COUNT,
      MAX_HEADER_SCAN_ROWS
    );

    return {
      company: uniqueSorted([
        ...pickColumnText(oaTable.rows, "公司简称"),
        ...pickColumnText(erpTable.rows, "区分公司简称")
      ]),
      dept1: uniqueSorted([
        ...pickColumnText(oaTable.rows, "一级部门"),
        ...pickColumnText(erpTable.rows, "一级部门")
      ]),
      dept2: uniqueSorted([
        ...pickColumnText(oaTable.rows, "二级部门"),
        ...pickColumnText(erpTable.rows, "二级部门")
      ])
    };
  } catch (error) {
    root?.console?.error?.("读取查询候选失败，查询弹窗将不显示补全下拉。", error);
    return EMPTY_QUERY_DIALOG_SUGGESTIONS;
  }
}
```

```ts
function writeDialogInitialState(root: ScrapVarianceGlobal, token: string, outputKind: OutputSheetKind | null): void {
  if (!outputKind) {
    return;
  }

  const activeSheet = root.Application?.ActiveSheet;
  if (!activeSheet) {
    return;
  }

  const payload: QueryDialogInitialPayload = {
    token,
    suggestions: EMPTY_QUERY_DIALOG_SUGGESTIONS
  };

  try {
    const state = readOutputQueryState(activeSheet);
    if (state) {
      payload.state = state;
    }
  } catch (error) {
    root.console?.error?.("读取输出表查询条件失败，查询弹窗将使用空条件。", error);
  }

  payload.suggestions = buildQueryDialogSuggestions(root);
  getStorage(root).setItem(buildDialogInitialStateKey(token), JSON.stringify(payload));
}
```

```js
  function readInitialPayload() {
    var storage = getStorage(false);
    var token = getToken();
    var raw;
    var parsed;

    if (!storage || !token) {
      return null;
    }

    raw = storage.getItem(getInitialStateKey());
    if (typeof raw !== "string" || !raw.trim()) {
      return null;
    }

    try {
      parsed = JSON.parse(raw);
    } catch (error) {
      return null;
    }

    if (!parsed || parsed.token !== token) {
      return null;
    }
    return parsed;
  }

    var payload = readInitialPayload();
```

```js
  function normalizeSuggestions(input) {
    return {
      company: input && Array.isArray(input.company) ? input.company : [],
      dept1: input && Array.isArray(input.dept1) ? input.dept1 : [],
      dept2: input && Array.isArray(input.dept2) ? input.dept2 : []
    };
  }

  function readSuggestions() {
    var payload = readInitialPayload();
    return normalizeSuggestions(payload && payload.suggestions);
  }

  function getMatchedOptions(value, suggestions) {
    var keyword = String(value || "").trim();
    var result = [];
    var index;
    var option;

    for (index = 0; index < suggestions.length; index += 1) {
      option = String(suggestions[index] || "");
      if (option && (!keyword || option.indexOf(keyword) >= 0)) {
        result.push(option);
      }
      if (result.length >= MAX_VISIBLE_OPTIONS) {
        break;
      }
    }
    return result;
  }

  function createAutocompleteDropdown() {
    var dropdown = document.createElement("div");
    dropdown.className = "autocomplete-menu";
    dropdown.setAttribute("role", "listbox");
    dropdown.style.display = "none";
    document.body.appendChild(dropdown);
    return dropdown;
  }

  function positionAutocompleteDropdown(input, dropdown) {
    var rect;
    if (typeof input.getBoundingClientRect !== "function") {
      return;
    }
    rect = input.getBoundingClientRect();
    dropdown.style.left = rect.left + "px";
    dropdown.style.top = rect.bottom + "px";
    dropdown.style.width = rect.width + "px";
  }

  function hideAutocompleteDropdown(dropdown) {
    dropdown.style.display = "none";
    while (dropdown.children.length > 0) {
      dropdown.removeChild(dropdown.children[0]);
    }
  }

  function renderAutocompleteOptions(dropdown, input, options) {
    var index;
    var item;

    hideAutocompleteDropdown(dropdown);
    if (!options.length) {
      return;
    }

    for (index = 0; index < options.length; index += 1) {
      item = document.createElement("div");
      item.className = "autocomplete-option";
      item.textContent = options[index];
      item.setAttribute("data-value", options[index]);
      item.setAttribute("role", "option");
      item.addEventListener("mousedown", function (event) {
        var value;
        if (event && typeof event.preventDefault === "function") {
          event.preventDefault();
        }
        value = event && event.target && typeof event.target.getAttribute === "function"
          ? event.target.getAttribute("data-value")
          : "";
        if (value != null) {
          input.value = value;
          hideAutocompleteDropdown(dropdown);
        }
      });
      dropdown.appendChild(item);
    }
    positionAutocompleteDropdown(input, dropdown);
    dropdown.style.display = "block";
  }

  function attachAutocomplete(inputId, suggestions) {
    var input = document.getElementById(inputId);
    var dropdown;

    if (!input || !suggestions.length) {
      return;
    }

    dropdown = createAutocompleteDropdown();
    input.addEventListener("focus", function () {
      renderAutocompleteOptions(dropdown, input, getMatchedOptions(input.value, suggestions));
    });
    input.addEventListener("input", function () {
      renderAutocompleteOptions(dropdown, input, getMatchedOptions(input.value, suggestions));
    });
    input.addEventListener("blur", function () {
      window.setTimeout(function () {
        hideAutocompleteDropdown(dropdown);
      }, 120);
    });
  }

  function initializeAutocomplete() {
    var suggestions = readSuggestions();
    attachAutocomplete("company", suggestions.company);
    attachAutocomplete("dept1", suggestions.dept1);
    attachAutocomplete("dept2", suggestions.dept2);
  }
```
